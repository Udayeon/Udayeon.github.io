# 코드
```cpp
#define _CRT_SECURE_NO_WARNINGS
#include <iostream>
using namespace std;

int a, b,max_num,min_num,r,result;

int solution(int num1, int num2) {
	while (num2 !=0) {
		r = num1 % num2; //6, 0
		num1 = num2; //num1 =18, 6
		num2 = r; //num2 = 6, 0
	}
	return num1;
}
int main() {
	cin >> a >> b;
	if (a >= b) { 
		max_num = a; 
		min_num = b; }
	else {
		max_num = b;
		min_num = a;
	}

	result= solution(max_num, min_num);
	cout << result <<'\n';
	cout << a * b / result;
}
```
