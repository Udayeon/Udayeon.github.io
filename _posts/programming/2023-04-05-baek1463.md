# (cpp)Baekjoon 1463번 문제 '1로 만들' - 다이나믹프로그래밍
* * *
[문제](https://www.acmicpc.net/problem/1463)   
   
# 코드
```cpp
#define _CRT_SECURE_NO_WARNINGS
#include <iostream>
using namespace std;

int a, b,max_num,min_num,r,result,t;

int solution(int num1, int num2) {
	if (num1 >= num2) {
		max_num = num1;
		min_num = num2;
	}
	else {
		max_num = num1;
		min_num = num2;
	}
	while (min_num !=0) {
		r = max_num % min_num; 
		max_num = min_num; 
		min_num = r;
	}
	return max_num;
}
int main() {
	cin >> t;
	for (int i = 0; i < t; i++) {
		cin >> a >>b;
		result = solution(a, b);
		cout << a * b / result <<'\n';
	}

}
```
