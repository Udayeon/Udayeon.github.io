# (cpp) Baekjoon 1260번 문제 'DFS와 BFS' - 그래프탐색,DFS,BFS
* * *
[문제](https://www.acmicpc.net/problem/1260)
# 풀이
```cpp
#define _CRT_SECURE_NO_WARNINGS
#include<iostream>
#include<vector>
#include<algorithm>
#include<queue>


using namespace std;
int n, m, v, from, to;
vector<int> g[1001];
queue<int> q;
bool check_dfs[1001];
bool check_bfs[1001];


void dfs(int v) {
	check_dfs[v] = true; //v정점 지나니까 true로 체크(v=1)
	cout << v << " "; //현재위치인 v 출력(아래에서 v가 넥스트로 업데이트 될 수 있음)
	//v에서 갈 수 있는 next탐색하기
	for (int i = 0; i < g[v].size(); i++)
	{
		int next = g[v][i]; //v에서 갈 수 있는 최소값 먼저 탐색
		if (!check_dfs[next]) {
			//만약 그곳이 첫 방문이면 그곳에서 갈 수 있는 next탐색
			dfs(next);
		}
	}
}

void bfs(int v) {
	q.push(v); //정점 q에 담음
	check_bfs[v] = true; //지났으니까 true;
	//q가 비워질 때 까지 반복
	while (!q.empty()) {
		int cur = q.front(); //현재 정점 값은 q의 맨앞 요소 q= [front , , ..];
		q.pop(); //먼저 들어간 원소 삭제해서 que비워줌.
		cout << cur << " "; //현재 정점 출력
		
		//현재 정점에서 갈 수 있는 next탐색
		for (int i = 0; i < g[cur].size(); i++) {
			int next = g[cur][i];
			if (!check_bfs[next]) {
				check_bfs[next] = true; //그곳이 첫 방문이면 방문 후 check.
				q.push(next); //q에 그 위치를 push하기.
			}

		}

	}
}


int main(void) {
	cin >> n >> m >> v;
	for (int i = 0; i < m; i++) {
		cin >> from >> to;
		g[from].push_back(to);
		g[to].push_back(from);
	}
	/*
	g[1] = 2 3 4;
	g[2] = 1 4;
	g[3] = 1 4;
	g[4] = 1 2 3;
	g=[[2,3,4],[1,4],[1,4],[1,2,3]]*/
	for (int i = 1; i <= n; i++) {
		sort(g[i].begin(), g[i].end());
	}
	dfs(v);
	cout << "\n";
	bfs(v);
	
	return 0;
}
```
